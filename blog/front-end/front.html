# 1. 防抖函数

# 在固定的时间内执行一次，如果被打断则重新开始计时。常见例子：坐电梯 LOL回城

  function debounced(fn, wait) {
    let timer = null;
    return function (...arg) {
      if (timer) {
        clearTimeout(timer)
      }
      timer = setTimeout(() => {
        fn.apply(this, arg);
      }, wait)
    }
  }

# 2. 节流函数

# 在规定的时间内只执行一次，不论触发频率多高。例如：过红绿灯 拖拽事件获取坐标

function throttle(fn, wait) {
    let start = 0;
    return function (...arg) {
      if (start === 0) {
        start = Date.now();
      } else {
        const now = Date.now();
        if (now - start >= wait) {
          fn.apply(this, arg);
          start = now;
        }
      }
    }
  }

# 3.深拷贝

# 便利方法：JSON.parse(JSON.stringify(data)) 可实现深拷贝，但是被拷贝数据中如果含有function 正则 Date Set Map等特殊对象则会丢失
# 基础数据类型：Number Boolean undefined Null String Symbol Bigint
# 引用数据类型：Object Array Function
# 浅拷贝是创建一个新对象，这个对象有着原始对象属性值的一份精确拷贝。如果属性是基本类型，拷贝的就是基本类型的值，如果属性是引用类型，拷贝的就是内存地址 ，
# 所以如果其中一个对象改变了这个地址，就会影响到另一个对象。
# 深拷贝是将一个对象从内存中完整的拷贝一份出来,从堆内存中开辟一个新的区域存放新对象,且修改新对象不会影响原对象。

# 总结：浅拷贝只复制指向某个对象的指针，而不复制对象本身，新旧对象还是共享同一块内存。但深拷贝会另外创造一个一模一样的对象，新对象跟原对象不共享内存，修改新对象不会改到原对象。

# 赋值 浅拷贝 深拷贝的区别

# 赋值：当我们把一个对象赋值给一个新的变量时，赋的其实是该对象的在栈中的地址，而不是堆中的数据。
# 也就是两个对象指向的是同一个存储空间，无论哪个对象发生改变，其实都是改变的存储空间的内容，因此，两个对象是联动的。
# 浅拷贝：重新在堆中创建内存，拷贝前后对象的基本数据类型互不影响，但拷贝前后对象的引用类型因共享同一块内存，会相互影响。
# 深拷贝：从堆内存中开辟一个新的区域存放新对象，对对象中的子对象进行递归拷贝,拷贝前后的两个对象互不影响。

# 总结：赋值之后，不论是基础数据类型还是引用数据类型都会互相影响。浅拷贝之后，基础数据类型不会影响，引用数据类型会互相影响。深拷贝后，基础数据类型和引用数据类型都不会受到影响。

# 基本类型直接存储在栈中,而引用类型的对象存储在堆中

function deepClone(obj) {
    // 判断克隆数据是不是基础数据类型
    // 不是则需要特殊处理
    if (typeof obj !== null && typeof obj === "object") {
      let cloneTarget = Array.isArray(obj) ? [] : {}; // 判断数据 是数组还是对象
      for (let key in obj) {
        if (obj.hasOwnProperty(key)) {
          cloneTarget[key] = deepClone(obj[key])
        }
      }
      return cloneTarget;
    } else {
      //基础数据类型直接返回
      return obj
    }
  }

# 4.实现instanceof方法

# Instanceof运算符的第一个变量是一个对象，暂时称为A；第二个变量一般是一个函数，暂时称为B。
# Instanceof的判断队则是：沿着A的__proto__这条线来找，同时沿着B的prototype这条线来找，如果两条线能找到同一个引用，即同一个对象，那么就返回true。如果找到终点还未重合，则返回false。

# 语法：object instanceof constructor

# 含义：object表示某个实例对象，constructor表示某个构造函数。它的作用是用于检查constructor.prototype是否存在于参数object的原型链上。

# A instanceof B

# 原理：本质做的就是递归A的原型链，看__proto__一层层向上查找直到null为止。看看有没有一处的__proto__与B.prototype相等,如果有则返回true,如果找到null都没有相等的，则返回false。
  function instanceof_(leftObject, rightConstructor) {
    // 前置判断 根据含义可知 leftObject是一个实例对象 rightConstructor是一个函数 不符合直接false
    if (typeof leftObject !== 'object' || leftObject === null || typeof rightConstructor !== 'function') {
      return false;
    }
    const leftProto = leftObject.__proto__;
    const rightPrototype = rightConstructor.prototype;
    while (true) {
      // 找到了顶端等于null 返回false
      if (leftProto === null) {
        return false;
      }
      // 找到了相等的 返回true
      if (leftProto === rightPrototype) {
        return true
      }
      // 一直顺着__proto__ 向上查找
      leftProto = leftProto.__proto__;
    }
  }
   function Person(){};
   let p = new Person();
   console.log(instanceof_(p,Person)); // true;
   console.log(instanceof_(new Number(1), Number));  // true;

# 5.模拟实现new操作符

# 当new一个构造函数时，得到的实例会继承构造器的属性和原型上的属性。使用new实例化大概过程：

# 1.创建一个简单的空JS对象。
# 2.给这个对象添加__proto__属性，并将该属性链接到构造函数的原型对象上。
# 3.调用构造函数,并将this绑定到新创建的对象。
# 4.如果构造函数没有显式返回一个对象，则返回新创建的对象。

# 注意：只能传入构造函数，不能传入class类

function new_(constructor, ...args) {
    // es5实现
    let obj = {};
    obj.__proto__ = constructor.prototype;
    // es6可合并第一二步
    //let obj = Object.create(constructor.prototype); // Object.create方法可以创建一个新对象并让新对象的__proto__指向已经存在的对象。
    const res = constructor.apply(obj, args);
    // 第四步会分为3种情况
    //1.如果构造函数没有显式return，那么就应该return新创建的obj对象。
    //2.如果构造函数返回的不是一个对象，那么就应该return新创建的obj对象。
    //3.如果构造函数显式返回了一个对象，如:{},function(){}，那么就是构造函数返回的这个对象。
    return res instanceof Object ? res : obj;
}
  function Person(name,age){
    this.name = name;
    this.age = age;
  }
  let p = new_(Person,'小红',6);
  let s = new Person('小红',6);
  console.log(s);  // {name:小红,age:6}
  console.log(p);  // {name:小红,age:6}

# 6.实现call()函数

# call/apply/bind 这三个方法都可以改变函数的this指向，可以让一个对象代理调用其他对象中的属性或方法。
# call和apply在改变函数的this指向后会立即执行，并且这两个方法接收的第一个参数都是需要调用其他对象方法的那个对象。
# 即B对象想调用A对象中的一个方法，则是A.methods.call(B,args);
# call和apply区别是接收参数的方式不一样，call是每个参数单独一个形参接收。即call(obj,arg1,arg2,arg3...)
# apply接收参数是以数组的形式接收。即apply(obj,[arg1,arg2,arg3...]);
# bind在改变函数的this指向后会返回一个函数，需要手动调用返回的函数。即A.methods.bind(B,args)();
# bind接收参数同call一样，即bind(obj,arg1,arg2,arg3...)。
 Function.prototype.myCall = function (obj) {
    // call的调用改变了this的指向 而this是函数，这是前提
    if (typeof this !== 'function') {
      throw new Error('type error')
    }
    //  处理接收到的参数
    let args = [...arguments].slice(1);
    // 判断是否传入参数 没有则绑定到window
    obj = obj || window;
    // 要调用方法，可以先将方法作为obj的一个属性再调用
    obj.fn = this;
    // 调用完方法之后，要保存方法的返回值并返回
    let result = obj.fn(...args);
    // 在原obj对象中，并没有fn属性，需要删除
    delete obj.fn;
    return result;
  }
  let animal = {
    say(name,sound){
      console.log(`${name}${sound}叫`);
    }
  }
  let cat = {}
  animal.say.myCall(cat,'猫','喵喵');  //猫喵喵叫

# 7.实现apply()函数

# apply函数除了在处理外部传进来的参数和call不一样之外，其他思路是一样的。
    Function.prototype.myApply = function (obj) {
    if (typeof this != 'function') {
      throw new Error('type error')
    }
    obj = obj || window;
    obj.fn = this;
    let result;
    if (arguments[1]) {
      result = obj.fn(...arguments[1]);
    } else {
      result = obj.fn();
    }
    delete obj.fn;
    return result;
  }
  let animal = {
    say(name, sound) {
      console.log(`${name}${sound}叫`);
    }
  }
  let dog = {}
  animal.say.myApply(dog, ['狗', '汪汪']); // 狗汪汪叫

# 8.实现bind()函数

# bind函数接收参数和call函数一样，不过返回值是一个函数
Function.prototype.myBind = function(obj){
    if(typeof this !=='function'){
      throw new Error('type error');
    }
    obj = obj || window;
    let that = this;
    // 获取参数
    let args = [...arguments].slice(1);
    //返回函数
    return function fn(){
      return that.apply(this instanceof fn?this:obj,args.concat(...arguments));
    }
  }
  let animal = {
    say(name,sound){
      console.log(`${name}${sound}叫`);
    }
  }
  let dog = {};
  animal.say.myBind(dog,"狗","汪汪")();  // 狗汪汪叫

# 9.实现Object.create()函数

# Object.create()函数可以接收两个参数 即 Object.create(proto,propertiesObject(可选参数));
# 作用是创建一个新的对象，使用现有对象来作为新创建对象的原型(prototype);
# proto参数需要为null或除基本类型包装以外的对象。如果不是这几类，则抛出一个TypeError。
  Object.myCreate = function(proto,propertiesObject=undefined){
    //判断proto的类型
    if(typeof proto !=='object' && typeof proto !=='function'){
      throw new Error('TypeError')
    }
    //propertiesObject不能为null
    if(propertiesObject===null){
      throw new Error('cannot convert undefined or null to object');
    }
    //利用原型链让新对象原型指向参数对象
    function F(){}
    F.prototype = proto;
    let obj = new F();  // 对象的隐式原型指向它构造函数的显式原型
    // 如果传入了第二个参数
    if(propertiesObject){
      Object.defineProperties(obj,propertiesObject);  // Object.defineProperties(对象,'属性名',描述对象);
    }
    // 支持创建没有原型的对象
    if(proto==null){
      obj.__proto__ = null;
    }
    return obj;
  }
